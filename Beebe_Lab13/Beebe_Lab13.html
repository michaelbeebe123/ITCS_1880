tipsByDay <- ddply(
  tips,
  "day",
  summarize,
  tip.mean=mean(tip),
  Length=NROW(tip),
  tip.sd=sd(tip),
  tfrac=qt(p=.90, df=Length-1),
  Lower=tip.mean - tfrac*tip.sd/sqrt(Length),
